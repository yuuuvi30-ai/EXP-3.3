<script type="text/babel">
class Person{
constructor(name,age){
this.name=name;
this.age=age;
}
getDetails(){
return `${this.name}, Age: ${this.age}`;
}
getRole(){
return "Person";
}
}
class Student extends Person{
constructor(name,age,grade){
super(name,age);
this.grade=grade;
}
getDetails(){
return `${this.name}, Age: ${this.age}, Grade: ${this.grade}`;
}
getRole(){
return "Student";
}
}
class Teacher extends Person{
constructor(name,age,subject){
super(name,age);
this.subject=subject;
}
getDetails(){
return `${this.name}, Age: ${this.age}, Subject: ${this.subject}`;
}
getRole(){
return "Teacher";
}
}
function App(){
const people=[
new Person("Anhad",19),
new Student("Yuvraj",19,"A"),
new Teacher("Amanpreet Kaur",25,"Full Stack")
];
return(
<div className="container">
<h1>Person Class Hierarchy</h1>
<div className="grid">
{people.map((person,index)=>(
<div className="card" key={index}>
<div className={`role ${person.getRole().toLowerCase()}`}>
{person.getRole()}
</div>
<h3>{person.name}</h3>
<p>{person.getDetails()}</p>
</div>
))}
</div>
</div>
);
}
const root=ReactDOM.createRoot(document.getElementById("root"));
root.render(<App/>);
</script>
